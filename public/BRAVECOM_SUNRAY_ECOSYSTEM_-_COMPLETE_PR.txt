### 3.3 RBAC Middleware Implementation

```typescript
// src/middleware/rbac-middleware.ts

import { NextRequest, NextResponse } from 'next/server';
import { verifyToken } from '@/lib/auth/jwt';
import { ROLE_PERMISSIONS, PERMISSIONS } from '@/lib/rbac/permission-codes';

export interface RBACConfig {
  requiredPermissions: string[];
  minLevel?: number;
  maxLevel?: number;
  excludeRoles?: string[];
}

export async function rbacMiddleware(
  request: NextRequest,
  config: RBACConfig
): Promise<NextResponse> {
  try {
    // Extract token from header
    const token = request.headers.get('authorization')?.replace('Bearer ', '');
    
    if (!token) {
      return NextResponse.json(
        { error: 'Authentication required', code: 'AUTH_MISSING' },
        { status: 401 }
      );
    }
    
    // Verify token and extract user info
    const user = await verifyToken(token);
    
    if (!user) {
      return NextResponse.json(
        { error: 'Invalid token', code: 'AUTH_INVALID' },
        { status: 401 }
      );
    }
    
    // Check role level
    const userLevel = getUserLevel(user.role);
    
    if (config.minLevel && userLevel < config.minLevel) {
      return NextResponse.json(
        { error: 'Insufficient access level', code: 'RBAC_LEVEL_LOW', required: config.minLevel, current: userLevel },
        { status: 403 }
      );
    }
    
    if (config.maxLevel && userLevel > config.maxLevel) {
      return NextResponse.json(
        { error: 'Access level too high', code: 'RBAC_LEVEL_HIGH', required: config.maxLevel, current: userLevel },
        { status: 403 }
      );
    }
    
    if (config.excludeRoles?.includes(user.role)) {
      return NextResponse.json(
        { error: 'Role excluded from this action', code: 'RBAC_ROLE_EXCLUDED' },
        { status: 403 }
      );
    }
    
    // Check permissions
    const userPermissions = ROLE_PERMISSIONS[user.role] || [];
    
    for (const requiredPerm of config.requiredPermissions) {
      const hasPermission = userPermissions.some(perm => {
        if (perm === '*:*:*') return true; // Root access
        if (perm.endsWith('*')) {
          const prefix = perm.slice(0, -1);
          return requiredPerm.startsWith(prefix);
        }
        return perm === requiredPerm;
      });
      
      if (!hasPermission) {
        return NextResponse.json(
          { error: 'Permission denied', code: 'RBAC_PERMISSION_DENIED', required: requiredPerm, role: user.role },
          { status: 403 }
        );
      }
    }
    
    // Attach user info to request headers for downstream use
    const requestHeaders = new Headers(request.headers);
    requestHeaders.set('x-user-id', user.id);
    requestHeaders.set('x-user-role', user.role);
    requestHeaders.set('x-user-level', userLevel.toString());
    
    return NextResponse.next({
      request: {
        headers: requestHeaders,
      },
    });
  } catch (error) {
    console.error('RBAC Middleware Error:', error);
    return NextResponse.json(
      { error: 'Authorization failed', code: 'RBAC_ERROR' },
      { status: 500 }
    );
  }
}

function getUserLevel(role: string): number {
  const levelMap: Record<string, number> = {
    GUEST: 1,
    REGISTERED: 2,
    INVESTOR: 3,
    VENDOR: 4,
    FINANCE: 5,
    COMPLIANCE: 6,
    ADMIN: 7,
    SUPER_ADMIN: 8,
    SYSTEM: 9,
    ROOT: 10,
  };
  
  return levelMap[role] || 0;
}
```

export const VENDOR_REVENUE_CONFIG = {
  VENDOR_SHARE: 0.85, // 85% to vendor
  PLATFORM_FEE: 0.15, // 15% to platform
  MIN_PRODUCTS_TO_ACTIVATE: 5,
  VENDOR_RETURN_RATE: 0.25, // 25% active return
} as const;

export function calculateRevenueShare(orderTotal: number): RevenueShareResult {
  const vendorShare = orderTotal * VENDOR_REVENUE_CONFIG.VENDOR_SHARE;
  const platformFee = orderTotal * VENDOR_REVENUE_CONFIG.PLATFORM_FEE;
  
  return {
    orderTotal,
    vendorShare,
    platformFee,
    vendorSharePercentage: VENDOR_REVENUE_CONFIG.VENDOR_SHARE,
    platformFeePercentage: VENDOR_REVENUE_CONFIG.PLATFORM_FEE,
  };
}

// src/lib/commission/admin-charges.ts

export const ADMIN_CONFIG = {
  ADMIN_CHARGE_PERCENT: 0.10, // 10% on commission only
  ADMIN_CHARGE_BASE: 'COMMISSION', // Not on principal or profit
  DEDUCTION_TIMING: 'PER_TRANSACTION', // Deducted per commission transaction
} as const;

export function calculateCommissionWithAdmin(
  investmentAmount: number,
  commissionRate: number
): CommissionResult {
  const grossCommission = investmentAmount * commissionRate;
  const adminCharge = grossCommission * ADMIN_CONFIG.ADMIN_CHARGE_PERCENT;
  const netCommission = grossCommission - adminCharge;
  
  return {
    investmentAmount,
    commissionRate,
    grossCommission,
    adminCharge,
    netCommission,
    adminChargePercentage: ADMIN_CONFIG.ADMIN_CHARGE_PERCENT,
  };
}

### 9.3 Trend Analysis Scoring

```typescript
// src/lib/product/trend-scoring.ts

export const TREND_SCORING = {
  // Market Demand Score (0-100)
  TREND_SCORE_WEIGHT: 0.40, // 40%
  SEARCH_VOLUME_WEIGHT: 0.30, // 30%
  GROWTH_RATE_WEIGHT: 0.30, // 30%
  
  // Thresholds
  HIGH_DEMAND_SCORE: 80,
  MEDIUM_DEMAND_SCORE: 60,
  LOW_DEMAND_SCORE: 40,
  
  // Data Collection
  MIN_DATA_HOURS: 24, // Minimum 24h before decision
  OPTIMAL_DATA_HOURS: 48, // Optimal 48h for best decisions
  MAX_DATA_HOURS: 168, // Max 7 days
  
  // Competition Levels
  LOW_COMPETITION_THRESHOLD: 10, // <10 competitors
  MEDIUM_COMPETITION_THRESHOLD: 50, // 10-50 competitors
  HIGH_COMPETITION_THRESHOLD: 50, // >50 competitors
} as const;

export function calculateMarketDemandScore(
  trendScore: number,
  searchVolume: number,
  growthRate: number
): number {
  const trendComponent = Math.min(trendScore, 100) * TREND_SCORING.TREND_SCORE_WEIGHT;
  const volumeComponent = Math.min((searchVolume / 10000) * 100, 100) * TREND_SCORING.SEARCH_VOLUME_WEIGHT;
  const growthComponent = Math.min((growthRate + 50) * 2, 100) * TREND_SCORING.GROWTH_RATE_WEIGHT;
  
  return trendComponent + volumeComponent + growthComponent;
}

    message: 'Please enter a valid 12-digit Aadhaar number',

    pattern: /^[A-Z]{5}[0-9]{4}[A-Z]{1}$/,
    message: 'Please enter a valid PAN number (ABCDE1234F)',
  },
  
  // IFSC
  IFSC: {

    pattern: /^[A-Z]{4}0[A-Z0-9]{6}$/,
    message: 'Please enter a valid 11-character IFSC code',
  },
  
  // Bank Account
  BANK_ACCOUNT: {
    minLength: 9,
    maxLength: 18,
    pattern: /^\d+$/,
    message: 'Account number must be 9-18 digits',
  },
  
  // Username
  USERNAME: {
    minLength: 6,
    maxLength: 20,
    pattern: /^[A-Za-z0-9_]+$/,
    message: 'Username must be 6-20 characters (letters, numbers, underscore only)',
  },
  
  // Password
  PASSWORD: {
    minLength: 8,
    requireUppercase: true,
    requireLowercase: true,
    requireNumber: true,
    requireSymbol: true,
    message: 'Password must be at least 8 characters with 1 uppercase, 1 lowercase, 1 number, and 1 symbol',
  },
  
  // OTP
  OTP: {
    length: 6,
    pattern: /^\d{6}$/,
    message: 'Please enter a valid 6-digit OTP',
  },
  
  // Referral Code
  REFERRAL_CODE: {
    minLength: 6,
    maxLength: 20,
    pattern: /^[A-Za-z0-9]+$/,
    message: 'Invalid referral code format',
  },
} as const;

export function validateField(field: string, value: string): ValidationResult {
  const rules = VALIDATION_RULES[field as keyof typeof VALIDATION_RULES];
  
  if (!rules) {
    return { valid: true };
  }
  
  if (rules.minLength && value.length < rules.minLength) {
    return { valid: false, error: `${field} must be at least ${rules.minLength} characters` };
  }
  
  if (rules.maxLength && value.length > rules.maxLength) {
    return { valid: false, error: `${field} must be at most ${rules.maxLength} characters` };
  }
  
  if (rules.pattern && !rules.pattern.test(value)) {
    return { valid: false, error: rules.message };
  }
  
  if (rules.length && value.length !== rules.length) {
    return { valid: false, error: `${field} must be exactly ${rules.length} characters` };
  }
  
  // Password specific validation
  if (field === 'PASSWORD') {
    if (rules.requireUppercase && !/[A-Z]/.test(value)) {
      return { valid: false, error: 'Password must contain at least 1 uppercase letter' };
    }
    if (rules.requireLowercase && !/[a-z]/.test(value)) {
      return { valid: false, error: 'Password must contain at least 1 lowercase letter' };
    }
    if (rules.requireNumber && !/[0-9]/.test(value)) {
      return { valid: false, error: 'Password must contain at least 1 number' };
    }
    if (rules.requireSymbol && !/[!@#$%^&*(),.?":{}|<>]/.test(value)) {
      return { valid: false, error: 'Password must contain at least 1 symbol' };
    }
  }
  
  // Aadhaar Luhn check
  if (field === 'AADHAAR' && rules.luhnCheck) {
    if (!luhnCheck(value)) {
      return { valid: false, error: 'Invalid Aadhaar number (checksum failed)' };
    }
  }
  
  return { valid: true };
}

function luhnCheck(value: string): boolean {
  // Implement Luhn algorithm for Aadhaar validation
  let sum = 0;
  let isEven = false;
  
  for (let i = value.length - 1; i >= 0; i--) {
    let digit = parseInt(value[i]);
    
    if (isEven) {
      digit *= 2;
      if (digit > 9) digit -= 9;
    }
    
    sum += digit;
    isEven = !isEven;
  }
  
  return sum % 10 === 0;
}

export interface ValidationResult {
  valid: boolean;
  error?: string;
}

### 9.2 Error Handling Middleware

```typescript
// src/middleware/error-handler.ts

import { NextRequest, NextResponse } from 'next/server';
import { createRegistrationError, RegistrationError } from '@/lib/registration/error-codes';

export async function errorHandler(
  request: NextRequest,
  next: () => Promise<NextResponse>
): Promise<NextResponse> {
  try {
    return await next();
  } catch (error) {
    console.error('Registration Error:', error);
    
    if (error instanceof RegistrationError) {
      return NextResponse.json(
        {
          success: false,
          error: {
            code: error.code,
            message: error.message,
            field: error.field,
            details: error.details,
          },
        },
        { status: error.statusCode }
      );
    }
    
    // Generic error
    const genericError = createRegistrationError(
      'SYSTEM_ERROR',
      'An unexpected error occurred. Please try again later.',
      undefined,
      [process.env.NODE_ENV === 'development' ? String(error) : undefined]
    );
    
    return NextResponse.json(
      {
        success: false,
        error: {
          code: genericError.code,
          message: genericError.message,
        },
      },
      { status: 500 }
    );
  }
}

## 1. OVERVIEW

### 1.1 Purpose
This document outlines the Role-Based Access Control (RBAC) system for the Sunray Ecosystem. The RBAC system provides granular access control across 10 levels, ensuring that users can only access resources and perform actions appropriate to their role.

### 1.2 Key Features
- **10 Access Levels**: From Guest (1) to Root (10)
- **Permission-Based**: Fine-grained permission codes
- **Resource-Level**: Control at module, endpoint, and data level
- **Dynamic**: Roles and permissions can be updated without code changes
- **Audit Trail**: All access attempts logged
- **Compliance**: Meets KYC/AML and data protection requirements

| Benefit | Description |
| :--- | :--- |
| **Security** | Prevents unauthorized access to sensitive data |

### 2.1 Architecture Diagram

```

│  │ Role Check   │  │ Permission   │  │ Resource Check       │  │
│  │ (Level 1-10) │  │ Validation   │  │ (Module/Endpoint)    │  │
│  └──────────────┘  └──────────────┘  └──────────────────────┘  │

              │  (200)    │       │  (403)    │
              └───────────┘       └───────────┘

### 2.2 RBAC Flow

```
1. User makes request with JWT token
        ↓
2. Middleware extracts token and verifies signature
        ↓
3. Extract user role and permissions from token
        ↓
4. Check required permissions for endpoint
        ↓
5. Check user role level (min/max)
        ↓
6. Check excluded roles (if any)
        ↓
7. Grant or deny access
        ↓
8. Log access attempt (audit trail)
```

## 6. IMPLEMENTATION GUIDE

### 6.1 RBAC Middleware Setup

```typescript
// src/middleware/rbac-middleware.ts

import { NextRequest, NextResponse } from 'next/server';
import { verifyToken } from '@/lib/auth/jwt';
import { ROLE_PERMISSIONS, ROLE_DEFINITIONS } from '@/lib/rbac/permission-codes';

export interface RBACConfig {
  requiredPermissions: string[];
  minLevel?: number;
  maxLevel?: number;
  excludeRoles?: string[];
  require2FA?: boolean;
}

export async function rbacMiddleware(
  request: NextRequest,
  config: RBACConfig
): Promise<NextResponse> {
  const startTime = Date.now();
  
  try {
    // 1. Extract token
    const token = request.headers.get('authorization')?.replace('Bearer ', '');
    
    if (!token) {
      await logAccessAttempt(request, 'DENIED', 'AUTH_MISSING');
      return NextResponse.json(
        { error: 'Authentication required', code: 'AUTH_MISSING' },
        { status: 401 }
      );
    }
    
    // 2. Verify token
    const user = await verifyToken(token);
    
    if (!user) {
      await logAccessAttempt(request, 'DENIED', 'AUTH_INVALID');
      return NextResponse.json(
        { error: 'Invalid token', code: 'AUTH_INVALID' },
        { status: 401 }
      );
    }
    
    // 3. Check 2FA requirement
    if (config.require2FA && !user.twoFactorEnabled) {
      await logAccessAttempt(request, 'DENIED', '2FA_REQUIRED');
      return NextResponse.json(
        { error: '2FA required for this action', code: '2FA_REQUIRED' },
        { status: 403 }
      );
    }
    
    // 4. Check role level
    const userLevel = ROLE_DEFINITIONS[user.role]?.level || 0;
    
    if (config.minLevel && userLevel < config.minLevel) {
      await logAccessAttempt(request, 'DENIED', 'RBAC_LEVEL_LOW', {
        required: config.minLevel,
        current: userLevel,
      });
      return NextResponse.json(
        { 
          error: 'Insufficient access level', 
          code: 'RBAC_LEVEL_LOW',
          required: config.minLevel,
          current: userLevel,
        },
        { status: 403 }
      );
    }
    
    if (config.maxLevel && userLevel > config.maxLevel) {
      await logAccessAttempt(request, 'DENIED', 'RBAC_LEVEL_HIGH', {
        required: config.maxLevel,
        current: userLevel,
      });
      return NextResponse.json(
        { 
          error: 'Access level too high', 
          code: 'RBAC_LEVEL_HIGH',
          required: config.maxLevel,
          current: userLevel,
        },
        { status: 403 }
      );
    }
    
    // 5. Check excluded roles
    if (config.excludeRoles?.includes(user.role)) {
      await logAccessAttempt(request, 'DENIED', 'RBAC_ROLE_EXCLUDED', {
        role: user.role,
      });
      return NextResponse.json(
        { error: 'Role excluded from this action', code: 'RBAC_ROLE_EXCLUDED' },
        { status: 403 }
      );
    }
    
    // 6. Check permissions
    const userPermissions = ROLE_PERMISSIONS[user.role] || [];
    
    for (const requiredPerm of config.requiredPermissions) {
      const hasPermission = userPermissions.some(perm => {
        if (perm === '*:*:*') return true; // Root access
        if (perm.endsWith('*')) {
          const prefix = perm.slice(0, -1);
          return requiredPerm.startsWith(prefix);
        }
        return perm === requiredPerm;
      });
      
      if (!hasPermission) {
        await logAccessAttempt(request, 'DENIED', 'RBAC_PERMISSION_DENIED', {
          required: requiredPerm,
          role: user.role,
          permissions: userPermissions,
        });
        return NextResponse.json(
          { 
            error: 'Permission denied', 
            code: 'RBAC_PERMISSION_DENIED',
            required: requiredPerm,
            role: user.role,
          },
          { status: 403 }
        );
      }
    }
    
    // 7. Attach user info to headers
    const requestHeaders = new Headers(request.headers);
    requestHeaders.set('x-user-id', user.id);
    requestHeaders.set('x-user-role', user.role);
    requestHeaders.set('x-user-level', userLevel.toString());
    
    // 8. Log successful access
    await logAccessAttempt(request, 'ALLOWED', 'SUCCESS', {
      duration: Date.now() - startTime,
    });
    
    return NextResponse.next({
      request: {
        headers: requestHeaders,
      },
    });
  } catch (error) {
    console.error('RBAC Middleware Error:', error);
    await logAccessAttempt(request, 'ERROR', 'RBAC_ERROR', {
      error: error instanceof Error ? error.message : 'Unknown error',
    });
    
    return NextResponse.json(
      { error: 'Authorization failed', code: 'RBAC_ERROR' },
      { status: 500 }
    );
  }
}

async function logAccessAttempt(
  request: NextRequest,
  decision: 'ALLOWED' | 'DENIED' | 'ERROR',
  reason: string,
  metadata?: Record<string, any>
): Promise<void> {
  // Log to audit_logs table
  const supabase = await createSupabaseServerClient();
  
  await supabase.from('audit_logs').insert({
    user_id: metadata?.userId || null,
    action: 'ACCESS_ATTEMPT',
    entity: 'RBAC',
    old_value: { decision, reason, ...metadata },
    new_value: null,
    ip_address: request.ip || request.headers.get('x-forwarded-for'),
    user_agent: request.headers.get('user-agent'),
    created_at: new Date().toISOString(),
  });
}
```

## 8. DATABASE SCHEMA

### 8.1 Roles Table

```sql
CREATE TABLE roles (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(50) UNIQUE NOT NULL,
    level INTEGER NOT NULL UNIQUE,
    description TEXT,
    can_upgrade BOOLEAN DEFAULT TRUE,
    can_downgrade BOOLEAN DEFAULT TRUE,
    requires_approval BOOLEAN DEFAULT FALSE,
    approval_level INTEGER,
    max_session_hours INTEGER,
    require_2fa BOOLEAN DEFAULT FALSE,
    audit_level VARCHAR(20) DEFAULT 'BASIC', -- NONE, BASIC, DETAILED, FULL
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_roles_name ON roles(name);
CREATE INDEX idx_roles_level ON roles(level);
CREATE INDEX idx_roles_active ON roles(is_active);
```

### 8.2 Permissions Table

```sql
CREATE TABLE permissions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    code VARCHAR(100) UNIQUE NOT NULL,
    module VARCHAR(50) NOT NULL,
    resource VARCHAR(50) NOT NULL,
    action VARCHAR(50) NOT NULL,
    description TEXT,
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_permissions_code ON permissions(code);
CREATE INDEX idx_permissions_module ON permissions(module);
CREATE INDEX idx_permissions_active ON permissions(is_active);
```

### 8.3 Role Permissions Table

```sql
CREATE TABLE role_permissions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    role_id UUID NOT NULL REFERENCES roles(id) ON DELETE CASCADE,
    permission_id UUID NOT NULL REFERENCES permissions(id) ON DELETE CASCADE,
    granted_by UUID REFERENCES users(id),
    granted_at TIMESTAMPTZ DEFAULT NOW(),
    expires_at TIMESTAMPTZ,
    UNIQUE(role_id, permission_id)
);

CREATE INDEX idx_role_permissions_role ON role_permissions(role_id);
CREATE INDEX idx_role_permissions_permission ON role_permissions(permission_id);
```

### 8.4 Audit Logs Table

```sql
CREATE TABLE audit_logs (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id),
    action VARCHAR(100) NOT NULL,
    entity VARCHAR(100) NOT NULL,
    entity_id UUID,
    old_value JSONB,
    new_value JSONB,
    ip_address INET,
    user_agent TEXT,
    rbac_decision VARCHAR(20), -- ALLOWED, DENIED, ERROR
    rbac_reason VARCHAR(100),
    rbac_metadata JSONB,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_audit_logs_user ON audit_logs(user_id);
CREATE INDEX idx_audit_logs_action ON audit_logs(action);
CREATE INDEX idx_audit_logs_created ON audit_logs(created_at);
CREATE INDEX idx_audit_logs_rbac ON audit_logs(rbac_decision);

-- Retention: 7 years (2555 days)
CREATE OR REPLACE FUNCTION cleanup_old_audit_logs()
RETURNS void AS $$
BEGIN
    DELETE FROM audit_logs WHERE created_at < NOW() - INTERVAL '2555 days';
END;
$$ LANGUAGE plpgsql;

## 9. SECURITY BEST PRACTICES

### 9.1 Principle of Least Privilege
- Grant minimum permissions required for role
- Regular permission audits (quarterly)
- Remove unused permissions immediately

### 9.2 Separation of Duties

- Compliance cannot verify their own KYC
- Admin changes require Super Admin approval

## 10. TESTING & AUDIT

### 10.1 RBAC Test Cases

```typescript
// src/tests/rbac/rbac.test.ts

describe('RBAC Access Control', () => {
  describe('Role Level Checks', () => {
    it('should deny Guest access to Investor endpoints', async () => {
      const response = await requestInvestmentCreate({ role: 'GUEST' });
      expect(response.status).toBe(403);
      expect(response.body.error.code).toBe('RBAC_LEVEL_LOW');
    });
    
    it('should allow Investor access to Investment endpoints', async () => {
      const response = await requestInvestmentCreate({ role: 'INVESTOR' });
      expect(response.status).toBe(200);
    });
    
    it('should deny Investor access to Admin endpoints', async () => {
      const response = await requestAdminUserManage({ role: 'INVESTOR' });
      expect(response.status).toBe(403);
      expect(response.body.error.code).toBe('RBAC_LEVEL_LOW');
    });
  });
  
  describe('Permission Checks', () => {
    it('should deny access without required permission', async () => {
      const response = await requestPayoutProcess({ role: 'INVESTOR' });
      expect(response.status).toBe(403);
      expect(response.body.error.code).toBe('RBAC_PERMISSION_DENIED');
    });
    
    it('should allow access with required permission', async () => {
      const response = await requestPayoutProcess({ role: 'FINANCE' });
      expect(response.status).toBe(200);
    });
  });
  
  describe('2FA Requirements', () => {
    it('should deny access when 2FA required but not enabled', async () => {
      const response = await requestInvestmentCreate({ 
        role: 'INVESTOR',
        twoFactorEnabled: false,
      });
      expect(response.status).toBe(403);
      expect(response.body.error.code).toBe('2FA_REQUIRED');
    });
    
    it('should allow access when 2FA enabled', async () => {
      const response = await requestInvestmentCreate({ 
        role: 'INVESTOR',
        twoFactorEnabled: true,
      });
      expect(response.status).toBe(200);
    });
  });
  
  describe('Audit Logging', () => {
    it('should log all access attempts', async () => {
      await requestInvestmentCreate({ role: 'GUEST' });
      
      const logs = await getAuditLogs({ action: 'ACCESS_ATTEMPT' });
      expect(logs.length).toBeGreaterThan(0);
      expect(logs[0].rbac_decision).toBe('DENIED');
    });
    
    it('should log successful access', async () => {
      await requestInvestmentCreate({ role: 'INVESTOR' });
      
      const logs = await getAuditLogs({ action: 'ACCESS_ATTEMPT' });
      expect(logs.some(l => l.rbac_decision === 'ALLOWED')).toBe(true);
    });
  });
});
```
